 #include <iostream>
 using namespace std;

 bool isZhishu(int num) {
    bool res = true;
    for(int i=2; i<=num-1; i++) {
        if(num%i == 0)
            res = false;
    }
    return res;
 }

 int nums(int n) {
    int count = 0;
    if(n <3 || n>=1000) {
        return -1;
    }
    int start = 2;
    while(start <= n/2) {
        if(isZhishu(n-start) && isZhishu(start))
            count++;
        start++;
    }
    return count;
 }

 int main() {
    int n;
    while (cin >> n) {
        int res ;
        res = nums(n);
        cout << res << endl;
    }
    return 0;
 }

#include <iostream>
#include <string>
#include <math.h>
using namespace std;

int code_index(string str) {
    int len = str.size();
    int index = 0;
    for(int i=0; i<len; i++,index++) {
        int n = str[i] - 'a';
        for(int j=0; j<4-i; j++) {
            index += n*pow(25,j);
        }
    }
    return index-1;
}

int main() {
    string in_str;
    int res;
    while(cin >> in_str) {
        res = code_index(in_str);
        cout << res <<endl;
    }
    return 0;
}
